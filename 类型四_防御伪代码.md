```
// ============================================================
// types: 防御体系共享的数据结构
// ============================================================

// 文档原始信息
struct Document:
    doc_id: str                          // 唯一标识
    content: str                         // 原始文本
    source: str                          // 来源（upload / api / sync）
    timestamp: int                       // 上传时间戳
    metadata: dict                       // 额外元数据（可选）

// 文档嵌入信息
struct DocumentEmbedding:
    doc_id: str
    vector: List[float]                  // n 维嵌入向量
    model_name: str                      // 使用的嵌入模型
    chunk_index: int | None              // 如果是分块后，所属 chunk 序号

// 正常文档统计信息（预处理阶段估计）
struct CleanStats:
    mu: List[float]                      // 均值向量, shape=(n,)
    sigma_inv: List[List[float]]         // 协方差逆矩阵, shape=(n, n)
    max_train_mahal: float               // 训练集最大马氏距离（用于归一化）
    max_train_lof_norm: float            // 训练集最大 LOF 归一化值
    reference_embeddings: List[List[float]]  // 参考嵌入集（用于 LOF/NNR）

// 异常检测结果
struct DetectionResult:
    doc_id: str
    is_anomaly: bool
    anomaly_score: float
    reason: str                          // 触发原因描述
    details: dict                        // 各子指标值（用于可解释性）

// chunk 相关
struct ChunkInfo:
    doc_id: str
    chunk_id: str
    chunk_index: int
    content: str
    embedding: List[float] | None

// 检索相关
struct RetrievedDoc:
    doc_id: str
    content: str
    embedding: List[float]
    score: float                         // 原始检索相似度
    trust_score: float | None            // SP6 处理后填充

// 检索分组
struct RetrievalGroup:
    group_id: int
    docs: List[RetrievedDoc]
    response: str | None                 // LLM 生成结果
    keywords: Set[str] | None            // 提取的关键词
    is_abstained: bool                   // 是否弃权

// NLI 结果
struct NLIResult:
    entailment: float                    // 蕴涵概率
    neutral: float                       // 中性概率
    contradiction: float                 // 矛盾概率

// 语义三元组（用于 SP7 语义图）
class SemanticTriple:
    head: str                            // 主体实体
    relation: str                        // 关系
    tail: str                            // 客体实体
```

## 1. SP1: EmbeddingAnomalyDetector（嵌入空间异常检测）

```
// ============================================================
// SP1: 检测上传文档在嵌入空间中是否为统计异常值
// 方法: 马氏距离 + LOF 加权融合
// 对应分析报告: 5.1 节
// 对应攻击: 节点1 PoisonedRAG 供应链投毒
// ============================================================

function EmbeddingAnomalyDetector(
    doc_embedding: List[float],
    clean_stats: CleanStats,
    reference_set: List[List[float]],     // 用于 LOF 计算的参考嵌入集
    threshold: float,                     // 融合得分阈值
    knn_k: int = 20,                    // LOF 的邻居数
    max_mahal: float | None,             // 训练集最大马氏距离（None 时自动计算）
    weight_mahal: float = 0.6,
    weight_lof: float = 0.4
) -> DetectionResult:

    // ----- 阶段 1: 马氏距离计算 -----
    // 公式: D_M(x) = sqrt((x - μ)^T Σ^{-1} (x - μ))
    diff = doc_embedding - clean_stats.mu                    // 偏离向量

    // 计算二次型: (x-μ)^T Σ^{-1} (x-μ)
    // 实现时使用 Cholesky 分解保证数值稳定性
    // temp = Σ^{-1/2} · diff → mahal_dist = ||temp||_2
    temp = cholesky_solve(clean_stats.sigma_inv, diff)       // Σ^{-1} @ diff
    mahal_dist = sqrt(dot(diff, temp))                       // 标量

    // 归一化马氏距离到 [0, 1]
    if max_mahal is not None:
        max_m = max_mahal
    else:
        max_m = clean_stats.max_train_mahal
    norm_mahal = mahal_dist / (max_m + EPSILON)

    // ----- 阶段 2: LOF 计算 -----
    // LOF(d) = (1/|N_k(d)|) * Σ_{d'∈N_k(d)} lrd(d') / lrd(d)
    // 其中 lrd(d) = 1 / ( (1/k) * Σ_{d'∈N_k(d)} ||d - d'||_2 + ε )

    // 2a. 找到 k 个最近邻
    distances = [euclidean(doc_embedding, ref) for ref in reference_set]
    neighbors_idx = argsort(distances)[1:k+1]                // 排除自身
    neighbors_dist = [distances[i] for i in neighbors_idx]

    // 2b. 计算局部可达密度 lrd(d)
    lrd_doc = 1.0 / (mean(neighbors_dist) + EPSILON)

    // 2c. 计算每个邻居的 lrd
    lrd_neighbors = []
    for idx in neighbors_idx:
        // 对每个邻居，找到它的 k 个最近邻
        neigh_dists = [euclidean(reference_set[idx], ref) for ref in reference_set]
        neigh_k_idx = argsort(neigh_dists)[1:k+1]
        neigh_k_dist = [neigh_dists[j] for j in neigh_k_idx]
        lrd_neigh = 1.0 / (mean(neigh_k_dist) + EPSILON)
        lrd_neighbors.append(lrd_neigh)

    // 2d. 计算 LOF 得分
    lof_score = mean(lrd_neighbors) / (lrd_doc + EPSILON)

    // LOF > 1: 文档比邻居更稀疏（异常）
    // LOF < 1: 文档比邻居更密集（正常聚类内部）
    norm_lof = abs(log2(lof_score)) / (clean_stats.max_train_lof_norm + EPSILON)

    // ----- 阶段 3: 融合决策 -----
    anomaly_score = weight_mahal * norm_mahal + weight_lof * norm_lof
    is_anomaly = anomaly_score > threshold

    // ----- 阶段 4: 构造结果 -----
    return DetectionResult(
        doc_id="",
        is_anomaly=is_anomaly,
        anomaly_score=anomaly_score,
        reason=(
            f"Mahalanobis={mahal_dist:.2f}(norm={norm_mahal:.2f}), "
            f"LOF={lof_score:.2f}(norm={norm_lof:.2f}), "
            f"fused={anomaly_score:.2f} vs threshold={threshold:.2f}"
        ) if is_anomaly else "normal",
        details={
            "mahalanobis_raw": mahal_dist,
            "mahalanobis_norm": norm_mahal,
            "lof_raw": lof_score,
            "lof_norm": norm_lof,
            "fused_score": anomaly_score,
            "threshold": threshold
        }
    )


// ============================================================
// 辅助函数: Cholesky 求解 Σ^{-1} @ diff
// 避免直接求逆矩阵（数值不稳定）
// ============================================================
function cholesky_solve(
    sigma_inv: List[List[float]],
    diff: List[float]
) -> List[float]:
    // 实际存储的是 Σ 而非 Σ^{-1}，用 Cholesky 分解求解线性系统
    // Σ · t = diff  →  t = Σ^{-1} · diff
    L = cholesky_decomposition(sigma)   // Σ = L · L^T
    y = forward_substitution(L, diff)   // L · y = diff
    t = backward_substitution(L.T, y)   // L^T · t = y
    return t


// ============================================================
// 辅助函数: 训练 CleanStats
// 在系统初始化时，用一批已知干净的文档计算统计量
// ============================================================
function train_clean_stats(
    clean_docs: List[DocumentEmbedding],
    shrinkage_rho: float = 0.3         // 收缩系数（处理高维协方差估计）
) -> CleanStats:

    N = len(clean_docs)
    dim = len(clean_docs[0].vector)

    // 均值向量
    mu = mean([doc.vector for doc in clean_docs], axis=0)

    // 中心化数据
    centered = [doc.vector - mu for doc in clean_docs]

    // 经验协方差矩阵 Σ_empirical = (1/(N-1)) * X^T · X
    sigma_empirical = (1.0 / (N - 1)) * (centered.T @ centered)

    // 收缩估计（shrinkage estimator）：
    // Σ̂ = (1 - ρ) · Σ_empirical + ρ · diag(Σ_empirical)
    // 作用：处理高维（dim >> N）时 Σ_empirical 条件数过大的问题
    sigma_shrunk = (1 - shrinkage_rho) * sigma_empirical
    for i in range(dim):
        sigma_shrunk[i][i] += shrinkage_rho * sigma_empirical[i][i]

    // 计算 Σ^{-1}（通过 Cholesky + 求逆）
    sigma_inv = invert_positive_definite(sigma_shrunk)

    // 计算训练集最大马氏距离和 LOF 归一化值
    max_mahal = 0.0
    max_lof_norm = 0.0
    for doc in clean_docs:
        diff = doc.vector - mu
        temp = cholesky_solve(sigma_shrunk, diff)
        mahal = sqrt(dot(diff, temp))
        max_mahal = max(max_mahal, mahal)

    // LOF 归一化值（在训练集上计算）
    for doc in clean_docs:
        // 简化: 计算 LOF 并取 log2 绝对值的最大值
        lof = compute_LOF_for_training(doc.vector, clean_docs)
        max_lof_norm = max(max_lof_norm, abs(log2(lof)))

    return CleanStats(
        mu=mu,
        sigma_inv=sigma_inv,
        max_train_mahal=max_mahal,
        max_train_lof_norm=max_lof_norm,
        reference_embeddings=[doc.vector for doc in clean_docs]
    )
```

## 2. SP2: ContentPerplexityAnalyzer（文档困惑度分析）

```
// ============================================================
// SP2: 基于语言模型困惑度检测文档是否包含非自然指令注入
// 方法: PPL + PD（前后差异）+ PM（最大值）三指标
// 对应分析报告: 5.2 节
// 对应攻击: 节点1 投毒指令注入（PoisonedRAG 的 I 子文本）
// ============================================================

function ContentPerplexityAnalyzer(
    doc: Document,
    lm_model,                           // 语言模型，提供 token_log_probs()
    alpha: float = 0.025,               // 假设检验显著性水平
    use_segmented: bool = True,         // 是否使用分段差异分析
    segment_ratio: float = 0.5,         // 前后段比例
    multi_model_voting: bool = False,   // 是否使用多模型集成（高级模式）
    lm_models_list: List | None = None  // 多模型列表
) -> DetectionResult:

    // ----- 阶段 1: 计算整体 PPL -----
    // PPL(d) = exp(-(1/T) * Σ_{t=1}^T log p(w_t | w_{<t}))
    tokens = tokenize(doc.content)
    log_probs = []

    for t in range(1, len(tokens)):
        context = tokens[:t]
        prob = lm_model.predict_next_prob(context, tokens[t])
        // prob = p(w_t | w_{<t})
        log_probs.append(log(prob + EPSILON))

    // 均值对数概率
    mean_log_prob = mean(log_probs)
    ppl_overall = exp(-mean_log_prob)

    // ----- 阶段 2: 分段 PPL 差异分析（基于 RAGuard）-----
    if use_segmented and len(tokens) >= 20:
        mid = int(len(tokens) * segment_ratio)

        // 前半段 PPL
        log_probs_pre = log_probs[:mid-1]
        ppl_pre = exp(-mean(log_probs_pre)) if log_probs_pre else 0.0

        // 后半段 PPL
        log_probs_post = log_probs[mid-1:]
        ppl_post = exp(-mean(log_probs_post)) if log_probs_post else 0.0

        // PD = PPL(pre) - PPL(post)
        pd_score = ppl_pre - ppl_post

        // PM = max(PPL(pre), PPL(post))
        pm_score = max(ppl_pre, ppl_post)
    else:
        pd_score = 0.0
        pm_score = ppl_overall

    // ----- 阶段 3: 滑动窗口局部 PPL 分析（捕捉短指令片段）-----
    // 某些投毒文档只有一段短指令异常，整体 PPL 被稀释
    window_size = min(50, len(tokens) // 4)
    stride = max(1, window_size // 2)

    local_ppls = []
    local_positions = []
    for start in range(0, len(tokens) - window_size + 1, stride):
        end = start + window_size
        window_log_probs = log_probs[start:end-1]
        if window_log_probs:
            local_ppl = exp(-mean(window_log_probs))
            local_ppls.append(local_ppl)
            local_positions.append((start, end))

    // 局部 PPL 统计
    max_local_ppl = max(local_ppls) if local_ppls else 0.0
    ppl_ratio = max_local_ppl / (ppl_overall + EPSILON)  // > 阈值表示有局部异常峰值

    // ----- 阶段 4: 多模型集成（高级模式）-----
    if multi_model_voting and lm_models_list is not None:
        // 使用多个不同架构的 LM 计算 PPL
        votes = []
        for other_lm in lm_models_list:
            other_ppl = compute_ppl_with_model(doc.content, other_lm)
            other_pd = compute_pd_with_model(doc.content, other_lm)
            is_flagged = (
                other_pd < percentile(empirical_pd_distribution, alpha) or
                other_pd > percentile(empirical_pd_distribution, 1 - alpha)
            )
            votes.append(is_flagged)

        // 多数投票
        multi_model_anomaly = sum(votes) > len(votes) // 2
    else:
        multi_model_anomaly = False

    // ----- 阶段 5: 决策逻辑 -----
    // 从经验分布中获取分位数阈值（需在训练时预计算）
    // 这里使用假设的预计算分位数值
    Q_alpha_PD = precomputed_quantiles["pd"][alpha]          // PD 下 α 分位数
    Q_1minus_alpha_PD = precomputed_quantiles["pd"][1-alpha]  // PD 上 α 分位数
    Q_1minus_alpha_PM = precomputed_quantiles["pm"][1-alpha]  // PM 上 α 分位数

    is_pd_anomaly = (pd_score <= Q_alpha_PD) or (pd_score >= Q_1minus_alpha_PD)
    is_pm_anomaly = pm_score >= Q_1minus_alpha_PM
    is_local_anomaly = ppl_ratio > precomputed_quantiles["ppl_ratio"][0.95]

    // 综合决策：任一指标异常即告警
    is_anomaly = is_pd_anomaly or is_pm_anomaly or is_local_anomaly
    if multi_model_voting:
        is_anomaly = is_anomaly or multi_model_anomaly

    // 构造原因
    reasons = []
    if is_pd_anomaly:
        reasons.append(f"PD异常(pd={pd_score:.2f})")
    if is_pm_anomaly:
        reasons.append(f"PM异常(pm={pm_score:.2f})")
    if is_local_anomaly:
        reasons.append(f"局部PPL峰值异常(ratio={ppl_ratio:.2f})")
    if multi_model_anomaly:
        reasons.append("多模型投票异常")

    return DetectionResult(
        doc_id=doc.doc_id,
        is_anomaly=is_anomaly,
        anomaly_score=float(is_anomaly),   // 或使用加权连续得分
        reason=" | ".join(reasons) if reasons else "normal",
        details={
            "ppl_overall": ppl_overall,
            "ppl_pre": ppl_pre if use_segmented else None,
            "ppl_post": ppl_post if use_segmented else None,
            "pd_score": pd_score,
            "pm_score": pm_score,
            "max_local_ppl": max_local_ppl,
            "ppl_ratio": ppl_ratio,
            "is_pd_anomaly": is_pd_anomaly,
            "is_pm_anomaly": is_pm_anomaly,
            "is_local_anomaly": is_local_anomaly
        }
    )


// ============================================================
// 辅助函数: 训练 PPL 经验分布
// 在干净文档集上预计算 PD 和 PM 的经验分布分位数
// ============================================================
function train_ppl_distribution(
    clean_docs: List[Document],
    lm_model,
    segment_ratio: float = 0.5
) -> dict:

    pd_values = []
    pm_values = []
    ppl_ratio_values = []

    for doc in clean_docs:
        tokens = tokenize(doc.content)
        if len(tokens) < 20:
            continue

        // 计算各 token 的对数概率
        log_probs = []
        for t in range(1, len(tokens)):
            context = tokens[:t]
            prob = lm_model.predict_next_prob(context, tokens[t])
            log_probs.append(log(prob + EPSILON))

        // 分段
        mid = int(len(tokens) * segment_ratio)
        log_probs_pre = log_probs[:mid-1]
        log_probs_post = log_probs[mid-1:]

        ppl_pre = exp(-mean(log_probs_pre)) if log_probs_pre else 0.0
        ppl_post = exp(-mean(log_probs_post)) if log_probs_post else 0.0

        pd_values.append(ppl_pre - ppl_post)
        pm_values.append(max(ppl_pre, ppl_post))

        // 滑动窗口
        window_size = min(50, len(tokens) // 4)
        stride = max(1, window_size // 2)
        local_ppls = []
        for s in range(0, len(tokens) - window_size + 1, stride):
            e = s + window_size
            w_log_probs = log_probs[s:e-1]
            if w_log_probs:
                local_ppls.append(exp(-mean(w_log_probs)))
        if local_ppls:
            ppl_ratio_values.append(max(local_ppls) / (exp(-mean(log_probs)) + EPSILON))

    return {
        "pd": compute_percentiles(pd_values, [0.025, 0.5, 0.975, 0.95]),
        "pm": compute_percentiles(pm_values, [0.025, 0.5, 0.975, 0.95]),
        "ppl_ratio": compute_percentiles(ppl_ratio_values, [0.95])
    }
```

## 3. SP3: CrossChunkCoherenceVerifier（跨块语义连贯性验证）

```
// ============================================================
// SP3: 检测跨 chunk 的语义断裂——攻击者将恶意指令碎片化
// 方法: 余弦连贯性 + LLM 条件概率 + 语义依赖图 三层融合
// 对应分析报告: 5.3 节
// 对应攻击: 节点2 语义混淆注入
// ============================================================

function CrossChunkCoherenceVerifier(
    chunks: List[ChunkInfo],             // 有序 chunks（来自同一文档）
    embed_model,                         // 嵌入模型
    llm_model,                           // 用于条件概率的 LLM
    threshold_cos: float = 0.5,          // 余弦连贯性阈值
    threshold_llm: float = 0.3,          // LLM 条件概率阈值
    coarse_threshold: float = 0.7,       // 触发 LLM 检查的粗阈值
    use_semantic_graph: bool = True,     // 是否使用语义图分析
    window_size: int = 3                 // 语义图滑动窗口大小
) -> List[DetectionResult]:              // 每个 chunk 边界一个结果

    if len(chunks) < 2:
        return []

    results = []

    // ----- 阶段 1: 逐对连贯性分析 -----
    for i in range(len(chunks) - 1):
        c_i = chunks[i]
        c_next = chunks[i + 1]

        // 1a. 余弦连贯性得分
        // Coh_cos(c_i, c_{i+1}) = (e_i · e_{i+1}) / (||e_i|| · ||e_{i+1}||)
        e_i = c_i.embedding if c_i.embedding else embed_model.encode(c_i.content)
        e_next = c_next.embedding if c_next.embedding else embed_model.encode(c_next.content)

        cos_coherence = cosine_similarity(e_i, e_next)    // 范围 [-1, 1]

        // 1b. LLM 条件概率连贯性（仅当余弦低于粗阈值时触发，节约开销）
        if cos_coherence < coarse_threshold:
            // 取 c_{i+1} 的前 K 个 token 计算条件概率
            prefix_tokens = tokenize(c_next.content)[:5]   // 前 5 个 token
            if prefix_tokens:
                cond_probs = []
                for t in range(1, len(prefix_tokens)):
                    context = prefix_tokens[:t]
                    // 以 c_i.content 为条件
                    full_context = c_i.content + " " + " ".join(context)
                    prob = llm_model.predict_next_prob(
                        tokenize(full_context),
                        prefix_tokens[t]
                    )
                    cond_probs.append(prob)

                // Coh_LM(c_i, c_{i+1}) = (1/K) * Σ log p(w_j | w_{<j}, c_i)
                llm_coherence = mean([log(p + EPSILON) for p in cond_probs])
                // 归一化到 [0, 1]（通过 sigmoid 或经验映射）
                llm_coherence_norm = 1.0 / (1.0 + exp(-llm_coherence))
            else:
                llm_coherence_norm = 1.0
        else:
            llm_coherence_norm = 1.0   // 余弦足够高，不做 LLM 检查

        // 1c. 单对融合决策
        is_coherent_pair = (cos_coherence >= threshold_cos) or (llm_coherence_norm >= threshold_llm)

        results.append(DetectionResult(
            doc_id=c_i.doc_id,
            is_anomaly=not is_coherent_pair,
            anomaly_score=1.0 - max(cos_coherence, llm_coherence_norm),
            reason=(
                f"chunk边界({i}-{i+1}): "
                f"cos={cos_coherence:.3f}, "
                f"llm={llm_coherence_norm:.3f}"
            ) if not is_coherent_pair else "normal",
            details={
                "chunk_i": i,
                "chunk_j": i + 1,
                "cos_coherence": cos_coherence,
                "llm_coherence": llm_coherence_norm,
                "is_llm_triggered": cos_coherence < coarse_threshold
            }
        ))

    // ----- 阶段 2: 语义依赖图分析（可选）-----
    if use_semantic_graph and len(chunks) >= 3:
        graph_anomalies = analyze_semantic_graph(
            chunks, embed_model, window_size
        )
        // 将图分析结果并入对应位置
        for ga in graph_anomalies:
            idx = ga["boundary_index"]
            if idx < len(results):
                results[idx].is_anomaly = results[idx].is_anomaly or ga["is_anomaly"]
                results[idx].reason += f" | graph: {ga['reason']}"

    return results


// ============================================================
// 辅助函数: 语义依赖图分析
// ============================================================
function analyze_semantic_graph(
    chunks: List[ChunkInfo],
    embed_model,
    window_size: int = 3
) -> List[dict]:

    anomalies = []

    // 对每个滑动窗口构建局部语义图
    for center in range(1, len(chunks) - 1):
        start = max(0, center - window_size // 2)
        end = min(len(chunks), center + window_size // 2 + 1)
        window_chunks = chunks[start:end]

        // 计算窗口内所有 chunk 对之间的余弦相似度
        N = len(window_chunks)
        sim_matrix = [[0.0] * N for _ in range(N)]
        for i in range(N):
            for j in range(i + 1, N):
                e_i = window_chunks[i].embedding or embed_model.encode(window_chunks[i].content)
                e_j = window_chunks[j].embedding or embed_model.encode(window_chunks[j].content)
                sim = cosine_similarity(e_i, e_j)
                sim_matrix[i][j] = sim
                sim_matrix[j][i] = sim

        // 边密度: 相似度 > 0.6 的边占比
        edge_count = sum(1 for i in range(N) for j in range(i+1, N) if sim_matrix[i][j] > 0.6)
        total_possible = N * (N - 1) / 2
        density = edge_count / total_possible if total_possible > 0 else 0.0

        // 碎片化攻击: 跨 chunk 边密度显著低于正常
        // 由于攻击 chunk 在语义上不连续，与其他 chunk 的相似度普遍低
        is_anomaly = density < 0.3       // 经验阈值

        anomalies.append({
            "boundary_index": center,
            "is_anomaly": is_anomaly,
            "graph_density": density,
            "reason": f"图密度={density:.2f} (< 0.3)"
        })

    return anomalies
```

## 4. SP4: TriggerRegionDetector（触发词紧凑区域检测）

```
// ============================================================
// SP4: 检测嵌入空间中 AgentPoison 的人为紧凑聚类
// 方法: LOF + NNR + 紧凑度 三指标融合
// 对应分析报告: 5.4 节
// 对应攻击: 节点3 AgentPoison 触发词检索劫持
// ============================================================

function TriggerRegionDetector(
    all_docs: List[DocumentEmbedding],          // 全量文档嵌入
    knn_k: int = 20,                           // kNN 参数
    nnr_threshold: float = 0.6,                // NNR 异常阈值
    lof_threshold: float = 2.0,                // LOF 阈值
    compactness_alpha: float = 0.3,            // 紧凑度融合权重
    use_dbscan: bool = True,                   // 是否使用聚类分析
    dbscan_eps: float = 0.5,                   // DBSCAN 半径
    dbscan_min_samples: int = 3                // DBSCAN 最小样本数
) -> List[DetectionResult]:                    // 每个文档一个结果

    N = len(all_docs)
    if N == 0:
        return []

    // 提取所有向量
    vectors = [doc.vector for doc in all_docs]

    // ----- 阶段 1: 全局 kNN 预计算（共享）-----
    // 构建 KD-Tree 或 Ball-Tree 加速邻居查询
    tree = build_kdtree(vectors)
    all_distances, all_indices = tree.query(vectors, k=knn_k + 1)  // 包含自身

    // ----- 阶段 2: 计算每个文档的 LOF -----
    // LOF(d) = (1/k) * Σ lrd(neighbor) / lrd(d)
    lrd_values = []
    for i in range(N):
        k_dist = all_distances[i][1:knn_k+1]        // 排除自身 (索引0)
        lrd_i = 1.0 / (mean(k_dist) + EPSILON)
        lrd_values.append(lrd_i)

    lof_scores = []
    for i in range(N):
        neighbor_lrd = [lrd_values[idx] for idx in all_indices[i][1:knn_k+1]]
        lof_i = mean(neighbor_lrd) / (lrd_values[i] + EPSILON)
        lof_scores.append(lof_i)

    // ----- 阶段 3: 计算每个文档的 NNR -----
    // NNR_k(d) = (avg_knn_dist(d)) / (global_avg_knn_dist)
    knn_dist_i = [all_distances[i][1:knn_k+1] for i in range(N)]
    avg_knn_dist_i = [mean(dists) for dists in knn_dist_i]
    global_avg_knn = mean(avg_knn_dist_i)

    nnr_scores = []
    for i in range(N):
        nnr = avg_knn_dist_i[i] / (global_avg_knn + EPSILON)
        nnr_scores.append(nnr)
    // NNR << 1: 文档的邻域异常紧凑（AgentPoison 特征）
    // NNR ≈ 1: 正常分布
    // NNR >> 1: 异常稀疏（离群点）

    // ----- 阶段 4: 聚类分析（可选）-----
    cluster_labels = None
    cluster_compactness = None
    if use_dbscan:
        // 使用 DBSCAN 对嵌入空间进行聚类
        cluster_labels = dbscan_cluster(vectors, eps=dbscan_eps, min_samples=dbscan_min_samples)

        // 计算每个聚类的紧凑度
        unique_labels = set(cluster_labels) - {-1}  // -1 表示噪声点
        cluster_compactness_map = {}
        for label in unique_labels:
            cluster_indices = [j for j, lbl in enumerate(cluster_labels) if lbl == label]
            if len(cluster_indices) < 2:
                continue
            cluster_vectors = [vectors[j] for j in cluster_indices]
            // Compactness(C) = (1/|C|²) * Σ Σ ||e_i - e_j||₂
            sum_dist = 0.0
            count = 0
            for a in range(len(cluster_vectors)):
                for b in range(a + 1, len(cluster_vectors)):
                    sum_dist += euclidean(cluster_vectors[a], cluster_vectors[b])
                    count += 1
            compactness = sum_dist / count if count > 0 else 0.0
            cluster_compactness_map[label] = compactness

        cluster_compactness = [cluster_compactness_map.get(lbl, -1.0) for lbl in cluster_labels]

    // ----- 阶段 5: 综合决策 -----
    results = []
    for i in range(N):
        lof = lof_scores[i]
        nnr = nnr_scores[i]

        // LOF 异常信号: |log2(LOF)| 越大越异常
        lof_anomaly_signal = abs(log2(lof + EPSILON))

        // NNR 异常信号: AgentPoison 文档的 NNR << 1
        nnr_anomaly_signal = max(0, 1.0 - nnr / nnr_threshold)

        // 紧凑度异常信号（如适用）
        compactness_signal = 0.0
        if cluster_compactness is not None and cluster_compactness[i] >= 0:
            // 聚类紧凑度越低（值越小）越可疑
            label = cluster_labels[i]
            if label != -1:
                compactness_signal = 1.0 / (cluster_compactness[i] + EPSILON)
                // 归一化
                all_compact = [c for c in cluster_compactness if c > 0]
                max_c = max(all_compact) if all_compact else 1.0
                compactness_signal = compactness_signal / (1.0 / max_c + EPSILON)

        // 融合得分
        // Score_AP = α₁ * |log(LOF)| + α₂ * max(0, 1 - NNR/τ) + α₃ * compactness
        anomaly_score = (
            0.4 * lof_anomaly_signal +
            0.4 * nnr_anomaly_signal +
            0.2 * compactness_signal
        )

        // 判定
        is_anomaly = (
            (abs(log2(lof)) > log2(lof_threshold)) or
            (nnr < nnr_threshold) or
            (compactness_signal > 0.8)
        )

        reasons = []
        if abs(log2(lof)) > log2(lof_threshold):
            reasons.append(f"LOF异常(lof={lof:.2f})")
        if nnr < nnr_threshold:
            reasons.append(f"NNR异常(nnr={nnr:.2f})")
        if compactness_signal > 0.8:
            reasons.append(f"紧凑度异常(compactness_signal={compactness_signal:.2f})")

        results.append(DetectionResult(
            doc_id=all_docs[i].doc_id,
            is_anomaly=is_anomaly,
            anomaly_score=anomaly_score,
            reason=" | ".join(reasons) if reasons else "normal",
            details={
                "lof_score": lof,
                "nnr_score": nnr,
                "compactness_signal": compactness_signal,
                "cluster_label": cluster_labels[i] if cluster_labels else None,
                "anomaly_score": anomaly_score
            }
        ))

    return results


// ============================================================
// 辅助函数: 全库周期性扫描（SP4 的典型使用模式）
// 作为后台离线任务运行，周期性地扫描向量数据库
// ============================================================
function periodic_scan_job(
    vector_db_client,                    // 向量数据库客户端
    scan_interval_hours: int = 24,       // 扫描间隔
    alert_threshold: float = 0.7         // 告警阈值
) -> None:

    while True:
        // 拉取全量文档嵌入
        all_embeddings = vector_db_client.fetch_all_embeddings()

        // 运行 SP4 检测
        results = TriggerRegionDetector(all_embeddings)

        // 筛选高置信度异常
        high_risk = [r for r in results if r.anomaly_score > alert_threshold]

        if high_risk:
            // 生成告警
            alert = {
                "type": "AGENTPOISON_SUSPECTED",
                "timestamp": now(),
                "triggered_docs": [r.doc_id for r in high_risk],
                "avg_score": mean([r.anomaly_score for r in high_risk]),
                "recommendation": "审查标为异常的文档，确认是否为 AgentPoison 攻击"
            }
            send_alert(alert)

            // 自动隔离高风险文档
            for r in high_risk:
                vector_db_client.mark_as_quarantined(r.doc_id)

        // 等待下一扫描周期
        sleep(scan_interval_hours * 3600)
```

## 5. SP5: RobustAggregationRetriever（鲁棒性聚合检索）

```
// ============================================================
// SP5: 通过隔离-聚合框架稀释单文档影响力，提供认证鲁棒性
// 方法: RobustRAG 隔离-聚合（关键词投票 + 解码置信度）
// 对应分析报告: 5.5 节
// 对应攻击: 节点3 检索劫持 + 全类型最终兜底
// ============================================================

function RobustAggregationRetriever(
    query: str,
    top_k_docs: List[RetrievedDoc],
    llm_model,                           // LLM 生成模型
    num_groups: int = 5,                 // 分组数 m
    alpha: float = 0.5,                  // 关键词阈值系数 α
    beta: int = 3,                       // 关键词阈值上限 β
    use_keyword_agg: bool = True,        // 使用关键词聚合
    use_decoding_agg: bool = False,      // 使用解码聚合（与关键词聚合互斥/互补）
    decoding_eta: float = 0.2,           // 解码置信度间隙 η
    enable_abstention: bool = True       // 允许组级弃权
) -> dict:

    // ----- 阶段 1: 文档分组（动态语义分组）-----
    // 改进: 基于语义相似度分配，而非简单顺序分组
    // 目标: 使同一攻击源的多个 chunk 尽可能在同一组（减少跨组污染）

    k = len(top_k_docs)
    group_size = ceil(k / num_groups)

    // 1a. 初始化组
    groups = [RetrievalGroup(group_id=j, docs=[], response=None, keywords=None, is_abstained=False)
              for j in range(num_groups)]

    // 1b. 降序排列文档（按检索得分）
    sorted_docs = sorted(top_k_docs, key=lambda d: d.score, reverse=True)

    // 1c. 交替分配（蛇形分配）：使每组包含不同排名区间的文档
    // 这样即使攻击者在某些排名位置有优势，也难以控制任意一组
    for idx, doc in enumerate(sorted_docs):
        if use_semantic_grouping:
            // 语义分组: 将语义最相似的文档分到同一组
            // 实现: 计算 doc 与各现有组的中心相似度，分配到最相似组
            best_group = find_most_similar_group(doc, groups, embed_model)
            groups[best_group].docs.append(doc)
        else:
            // 简单交替分组
            group_id = idx % num_groups
            groups[group_id].docs.append(doc)

    // ----- 阶段 2: 逐组独立推理（隔离）-----
    // y_j = LLM(Q ⊕ G_j)
    for group in groups:
        if not group.docs:
            group.is_abstained = True
            continue

        // 构造组上下文
        context = format_context(group.docs)
        prompt = build_prompt(query, context)

        // LLM 生成
        response = llm_model.generate(prompt, max_tokens=512)

        // 弃权检测
        if enable_abstention:
            if is_refusal_response(response) or is_empty_response(response):
                group.is_abstained = True
                continue

        group.response = response

        // 提取关键词（用于关键词聚合）
        if use_keyword_agg:
            group.keywords = extract_keywords(response)

    // ----- 阶段 3: 聚合 -----
    non_abstained = [g for g in groups if not g.is_abstained]
    n = len(non_abstained)

    if n == 0:
        return {
            "final_response": None,
            "is_robust": False,
            "reason": "所有组均弃权",
            "details": {"n_abstained": num_groups}
        }

    if use_keyword_agg:
        // 3a. 关键词聚合
        // Freq(w) = Σ_{j=1}^m I(w ∈ K_j)
        keyword_counts = {}
        for group in non_abstained:
            for kw in group.keywords:
                keyword_counts[kw] = keyword_counts.get(kw, 0) + 1

        // 阈值: μ = min(α · n, β)
        mu = min(alpha * n, beta)

        // 稳健关键词: Freq(w) ≥ μ
        robust_keywords = {
            kw: count for kw, count in keyword_counts.items()
            if count >= mu
        }

        // 构建最终回答
        final_response = format_keyword_response(robust_keywords)

        return {
            "final_response": final_response,
            "robust_keywords": robust_keywords,
            "keyword_counts": keyword_counts,
            "mu_threshold": mu,
            "n_groups": n,
            "n_abstained": num_groups - n,
            "is_robust": len(robust_keywords) > 0
        }

    elif use_decoding_agg:
        // 3b. 解码聚合（替代方案，适用于生成式任务）
        // 每一步: avg_probs = mean(p_t^{(1)}, ..., p_t^{(m)})
        // 选择: top1 - top2 > η 时选 top1，否则弃权

        final_tokens = []
        for step in range(max_decode_steps):
            // 收集各组的下一步概率分布
            all_probs = []
            for group in non_abstained:
                if group.response is not None:
                    probs = get_next_token_probs(llm_model, group.response, step)
                    all_probs.append(probs)

            if not all_probs:
                break

            // 平均概率
            avg_probs = mean_prob_distributions(all_probs)

            // 取 Top-2 概率
            top2 = sorted(avg_probs.items(), key=lambda x: x[1], reverse=True)[:2]
            confidence_gap = top2[0][1] - top2[1][1]

            if confidence_gap > decoding_eta:
                final_tokens.append(top2[0][0])
            else:
                break   // 置信度不足，停止生成

            "final_response": final_response,
            "decoding_steps": len(final_tokens),

function extract_keywords(text: str) -> Set[str]:
    // 使用简单规则或 NLP 工具提取关键词
    // 1. 分词 + 去除停用词
```

## 6. SP6: PostRetrievalVerifier（后检索一致性验证）

```
// ============================================================
// SP6: LLM 调用前验证检索结果的 Q/C/M 三维一致性
// 方法: SeConRAG CAF（Conflict-Aware Filtering）框架
// 对应分析报告: 5.6 节
// 对应攻击: 全类型 — 检索阶段最后一层防线
// ============================================================

function PostRetrievalVerifier(
    query: str,
    retrieved_docs: List[RetrievedDoc],
    cross_encoder_model,                 // 交叉编码器（Query, Doc → 相关性评分）
    embed_model,                         // 嵌入模型（用于文档间余弦）
    nli_model,                           // NLI 模型（蕴涵/矛盾检测）
    llm_base_response: str | None,       // LLM 无上下文的基础回答（可选）
    trust_threshold: float = 0.5,        // 信任得分阈值
    lazy_cross_encoder_threshold_low: float = 0.6,   // 触发交叉编码器的下界
    lazy_cross_encoder_threshold_high: float = 0.9,  // 触发交叉编码器的上界
    top_k_nli_check: int = 3             // 仅对 Top-K 差异文档做 NLI
) -> tuple:                              // (trusted, filtered_out, scores)

    k = len(retrieved_docs)
    if k == 0:
        return [], [], []

    // 提取文本列表和嵌入
    doc_texts = [doc.content for doc in retrieved_docs]
    doc_embeddings = []
    for doc in retrieved_docs:
        if doc.embedding:
            doc_embeddings.append(doc.embedding)
        else:
            doc_embeddings.append(embed_model.encode(doc.content))

    // ----- 阶段 1: 查询-文档一致性（Q-Consistency）-----
    // 嵌入余弦得分（轻量级）
    q_embedding = embed_model.encode(query)
    cos_scores = [cosine_similarity(q_embedding, de) for de in doc_embeddings]

    // 交叉编码器评分（仅在余弦得分模糊区域触发，延迟执行策略）
    // 在 0.6~0.9 范围: 余弦不确定性最大，交叉编码器最有价值
    cross_scores = []
    for i in range(k):
        if lazy_cross_encoder_threshold_low <= cos_scores[i] <= lazy_cross_encoder_threshold_high:
            // 余弦模糊 → 用交叉编码器精排
            cross_score = cross_encoder_model.predict(query, doc_texts[i])
            cross_scores.append(cross_score)
        else:
            // 余弦明确（极低或极高）→ 直接用余弦
            cross_scores.append(cos_scores[i])

    // 对齐得分（Final Q-Consistency）
    align_scores = [
        0.3 * cos_scores[i] + 0.7 * cross_scores[i]
        for i in range(k)
    ]

    // ----- 阶段 2: 文档间一致性（C-Consistency）-----
    // Sim_ij = cosine(E(d_i), E(d_j))
    sim_matrix = [[0.0] * k for _ in range(k)]
    for i in range(k):
        for j in range(i + 1, k):
            sim = cosine_similarity(doc_embeddings[i], doc_embeddings[j])
            sim_matrix[i][j] = sim
            sim_matrix[j][i] = sim

    // 平均文档间相似度 S̄ = (2/k(k-1)) * Σ_{i<j} Sim_ij
    all_sims = [sim_matrix[i][j] for i in range(k) for j in range(i+1, k)]
    overall_mean_sim = mean(all_sims)

    // 单个文档的偏离度 Dev_i = S̄_i - S̄
    // S̄_i = mean({Sim_ij | j ≠ i})
    dev_scores = []
    for i in range(k):
        sims_to_others = [sim_matrix[i][j] for j in range(k) if j != i]
        mean_sim_i = mean(sims_to_others)
        dev_i = mean_sim_i - overall_mean_sim
        dev_scores.append(dev_i)

    // 归一化为 [0, 1] 的惩罚因子: 负偏离越大（越不一致），因子越小
    min_dev = min(dev_scores)
    max_dev = max(dev_scores)
    dev_range = max_dev - min_dev if max_dev > min_dev else 1.0
    consistency_penalties = [
        1.0 - (max(0, -dev) / max(0.001, -min_dev)) * 0.5
        for dev in dev_scores
    ]

    // ----- 阶段 3: 模型一致性（M-Consistency, 可选）-----
    fact_scores = [1.0] * k
    if llm_base_response is not None and top_k_nli_check > 0:
        // 仅对 Dev_i 最大的 Top-K 文档做 NLI 检查（节约计算）
        dev_ranked = sorted(
            enumerate(dev_scores),
            key=lambda x: x[1]    // 从小到大（负偏离越大越优先检查）
        )
        for rank_idx in range(min(top_k_nli_check, k)):
            doc_idx = dev_ranked[rank_idx][0]

            // NLI(d, LLM_base(Q))
            nli_input = (doc_texts[doc_idx], llm_base_response)
            nli_result = nli_model.predict(nli_input)

            // FactConsist = 1.0(蕴涵) / 0.3(中性) / 0.0(矛盾)
            if nli_result.entailment > nli_result.contradiction:
                fact_scores[doc_idx] = nli_result.entailment
            elif nli_result.contradiction > nli_result.entailment:
                fact_scores[doc_idx] = 0.0
            else:
                fact_scores[doc_idx] = 0.3

    // ----- 阶段 4: 信任得分融合 -----
    // Trust(d) = Align_cross(Q,d) * (0.5 + 0.5 * consistency_penalty) * FactConsist
    trust_scores = []
    for i in range(k):
        trust = (
            align_scores[i] *
            consistency_penalties[i] *
            fact_scores[i]
        )
        trust_scores.append(trust)

    // 动态阈值: 基于信任得分的统计分布
    if k >= 3:
        // 使用均值 - 0.5 * 标准差 作为自适应阈值
        mean_trust = mean(trust_scores)
        std_trust = std(trust_scores)
        dynamic_threshold = max(
            trust_threshold,
            mean_trust - 0.5 * std_trust
        )
    else:
        dynamic_threshold = trust_threshold

    // ----- 阶段 5: 过滤 -----
    trusted = []
    filtered_out = []
    for i, doc in enumerate(retrieved_docs):
        doc.trust_score = trust_scores[i]
        if trust_scores[i] >= dynamic_threshold:
            trusted.append(doc)
        else:
            filtered_out.append(doc)

    return trusted, filtered_out, {
        "trust_scores": trust_scores,
        "threshold": dynamic_threshold,
        "align_scores": align_scores,
        "dev_scores": dev_scores,
        "fact_scores": fact_scores,
        "consistency_penalties": consistency_penalties,
        "doc_rankings": [doc.doc_id for doc in retrieved_docs]
    }
```

## 7. SP7: SemanticDependencyGraphAnalyzer（语义依赖图分析）

```
// ============================================================
// SP7: 跨文档碎片化攻击检测 — 分析文档间的语义依赖关系图
// 方法: 构建 EIRE 语义三元组 → 跨文档依赖图 → 图异常检测
// 对应分析报告: 5.3.2 节（方法3: 语义依赖图）
// 对应攻击: 节点2 跨文档语义混淆注入
// ============================================================

function SemanticDependencyGraphAnalyzer(
    doc_group: List[Document],           // 同一批次/来源的文档组
    nlp_extractor,                       // 实体-意图-关系提取器（EIRE）
    embed_model,                         // 嵌入模型
    density_threshold: float = 0.25,     // 边密度阈值
    anomaly_edge_ratio: float = 0.4,     // 异常边比例阈值
    use_cross_doc_edges: bool = True,    // 是否构建跨文档边
    extract_top_k_triples: int = 20      // 每文档提取的最大三元组数
) -> DetectionResult:

    N = len(doc_group)
    if N < 2:
        return DetectionResult(
            is_anomaly=False, reason="文档数 < 2，不适合图分析"
        )

    // ----- 阶段 1: 提取每个文档的语义三元组 -----
    // 三元组: (E, I, R) — 实体, 意图, 关系
    all_triples = []                     // [(doc_id, triple)]
    for doc in doc_group:
        // 使用 NLP 提取语义三元组
        triples = nlp_extractor.extract_semantic_triples(
            doc.content,
            top_k=extract_top_k_triples
        )
        for triple in triples:
            all_triples.append((doc.doc_id, triple))

    // ----- 阶段 2: 构建语义依赖图 -----
    // G = (V, E), V 是实体节点, E 是依赖边

    // 2a. 收集所有实体作为节点
    entity_set = set()
    for doc_id, triple in all_triples:
        entity_set.add(triple.head)
        entity_set.add(triple.tail)

    entity_list = list(entity_set)
    entity_to_idx = {e: i for i, e in enumerate(entity_list)}
    M = len(entity_list)

    // 2b. 构建邻接矩阵
    adj = [[0.0] * M for _ in range(M)]
    edge_doc_map = {}                    // (i,j) -> set of doc_ids
    for doc_id, triple in all_triples:
        i = entity_to_idx[triple.head]
        j = entity_to_idx[triple.tail]
        adj[i][j] += 1.0
        adj[j][i] += 1.0                 // 无向图（或保持有向）
        key = (i, j) if i < j else (j, i)
        if key not in edge_doc_map:
            edge_doc_map[key] = set()
        edge_doc_map[key].add(doc_id)

    // ----- 阶段 3: 计算图度量 -----
    // 3a. 整体边密度
    // ρ(G) = 2|E| / (|V|(|V|-1))
    edge_count = sum(1 for i in range(M) for j in range(i+1, M) if adj[i][j] > 0)
    total_possible = M * (M - 1) / 2 if M > 1 else 1
    graph_density = edge_count / total_possible

    // 3b. 跨文档边比例
    cross_doc_edges = 0
    total_edges_with_docs = 0
    for key, doc_set in edge_doc_map.items():
        if len(doc_set) > 1:            // 多个文档指向同一实体对
            cross_doc_edges += 1
        total_edges_with_docs += 1

    cross_doc_ratio = cross_doc_edges / total_edges_with_docs if total_edges_with_docs > 0 else 0.0

    // 碎片化攻击特征:
    // - 正常文档组: 实体间高密度连通，边密度高
    // - 碎片化攻击: 不同文档的实体之间缺乏自然连接，边密度低
    // - 但攻击的"跨文档边比例"可能很高（攻击文档共享同一攻击目标实体）

    // 3c. 单文档子图特性
    doc_subgraph_metrics = {}
    for doc in doc_group:
        // 提取该文档的实体子集
        doc_entities = set()
        for d_id, triple in all_triples:
            if d_id == doc.doc_id:
                doc_entities.add(triple.head)
                doc_entities.add(triple.tail)

        if len(doc_entities) < 2:
            continue

        // 该文档内部的边数
        internal_edges = 0
        internal_possible = 0
        ent_list = list(doc_entities)
        for a in range(len(ent_list)):
            for b in range(a + 1, len(ent_list)):
                i = entity_to_idx[ent_list[a]]
                j = entity_to_idx[ent_list[b]]
                internal_possible += 1
                if adj[i][j] > 0:
                    internal_edges += 1

        internal_density = internal_edges / internal_possible if internal_possible > 0 else 0.0

        // 该文档实体与其他文档实体的连接数（跨文档边）
        cross_edges = 0
        for ent in doc_entities:
            i = entity_to_idx[ent]
            for j in range(M):
                if j != i and adj[i][j] > 0:
                    # 检查 j 是否属于其他文档
                    for d_id2, triple2 in all_triples:
                        if d_id2 != doc.doc_id and (triple2.head == entity_list[j] or triple2.tail == entity_list[j]):
                            cross_edges += 1
                            break
        // 简化: cross_edges 表示该文档实体与其他文档实体间的连接

        doc_subgraph_metrics[doc.doc_id] = {
            "entity_count": len(doc_entities),
            "internal_density": internal_density,
            "cross_edges": cross_edges
        }

    // ----- 阶段 4: 异常判定 -----
    // 碎片化攻击的特征组合:
    //   1. 全局边密度低 (< density_threshold)
    //   2. 存在异常文档: 内部密度接近0 但 跨文档边比例高
    //   3. 攻击文档的实体集小且孤立

    is_global_sparse = graph_density < density_threshold

    // 检测异常文档
    anomalous_docs = []
    for doc_id, metrics in doc_subgraph_metrics.items():
        is_suspicious = (
            metrics["internal_density"] < density_threshold and
            metrics["cross_edges"] > 0 and
            metrics["entity_count"] <= 3    // 攻击文档通常只有少量实体
        )
        if is_suspicious:
            anomalous_docs.append(doc_id)

    is_anomaly = is_global_sparse and len(anomalous_docs) >= 2

    return DetectionResult(
        doc_id="|".join(doc.doc_id for doc in doc_group),   // 组级别
        is_anomaly=is_anomaly,
        anomaly_score=graph_density if not is_anomaly else 1.0 - graph_density,
        reason=(
            f"图密度={graph_density:.3f}(<{density_threshold}), "
            f"跨文档边比例={cross_doc_ratio:.2f}, "
            f"异常文档数={len(anomalous_docs)}"
        ) if is_anomaly else f"图密度={graph_density:.3f}, 正常",
        details={
            "graph_density": graph_density,
            "cross_doc_ratio": cross_doc_ratio,
            "entity_count": M,
            "edge_count": edge_count,
            "anomalous_docs": anomalous_docs,
            "doc_metrics": doc_subgraph_metrics
        }
    )
```

## 8. Pipeline 管线编排

```
// ============================================================
// pipeline.py: 防御体系主入口，将 SP1-SP7 编排为完整的拦截管线
// 支持 5 种配置模式（Config-1 ~ Config-5）
// 对应分析报告: 第 6 节
// ============================================================

// 配置枚举
enum PipelineConfig:
    CONFIG_1_FAST = "config_1"           // SP1 + SP2: 快速基础
    CONFIG_2_STANDARD = "config_2"       // SP1 + SP2 + SP3: 标准安全
    CONFIG_3_FULL_UPLOAD = "config_3"    // SP1 + SP2 + SP3 + SP4: 全面上传防护
    CONFIG_4_RETRIEVAL_ONLY = "config_4" // SP5 + SP6: 仅检索阶段
    CONFIG_5_MAX = "config_5"            // SP1~SP7 全开: 最高安全等级


// 主防御管线
function DefensePipeline(
    config: PipelineConfig,
    init_params: dict                    // 各节点的初始参数
) -> None:

    // 初始化各防御节点（按需懒加载）
    detectors = {}
    stats = {}

    // ---- 管线主入口: upload_document ----
    // 文档上传时调用
    async function upload_document(
        doc: Document
    ) -> dict:

        result = {
            "doc_id": doc.doc_id,
            "allowed": True,             // 是否允许入库
            "alerts": [],                // 告警列表
            "details": {}                // 各节点详细结果
        }

        // ---- 上传阶段检查 ----

        // SP1: 嵌入空间异常检测
        if config in [CONFIG_1_FAST, CONFIG_2_STANDARD, CONFIG_3_FULL_UPLOAD, CONFIG_5_MAX]:
            if "sp1" not in detectors:
                detectors["sp1"] = EmbeddingAnomalyDetector(**init_params.get("sp1", {}))
                stats["sp1"] = load_clean_stats(init_params["clean_stats_path"])

            doc_emb = embed_document(doc.content)
            sp1_result = detectors["sp1"](
                doc_emb,
                stats["sp1"],
                stats["sp1"].reference_embeddings
            )

            result["details"]["sp1"] = sp1_result
            if sp1_result.is_anomaly:
                result["alerts"].append({
                    "node": "SP1",
                    "reason": sp1_result.reason,
                    "score": sp1_result.anomaly_score
                })
                // Config-5: 一票否决
                if config == CONFIG_5_MAX:
                    result["allowed"] = False

        // SP2: PPL 困惑度分析
        if config in [CONFIG_1_FAST, CONFIG_2_STANDARD, CONFIG_3_FULL_UPLOAD, CONFIG_5_MAX]:
            if "sp2" not in detectors:
                detectors["sp2"] = ContentPerplexityAnalyzer(**init_params.get("sp2", {}))

            sp2_result = detectors["sp2"](doc)
            result["details"]["sp2"] = sp2_result
            if sp2_result.is_anomaly:
                result["alerts"].append({
                    "node": "SP2",
                    "reason": sp2_result.reason
                })
                if config == CONFIG_5_MAX:
                    result["allowed"] = False

        // SP3: 跨块连贯性（需要分块后执行）
        // 注意: SP3 在分块后调用，不是在原始文档上
        // 因此这里只记录 SP3 需要在分块阶段执行
        result["pending_checks"] = ["SP3"] if config in [
            CONFIG_2_STANDARD, CONFIG_3_FULL_UPLOAD, CONFIG_5_MAX
        ] else []

        return result


    // ---- 管线入口: process_chunks ----
    // 文档分块后调用（对 SP3 友好）
    async function process_chunks(
        doc_id: str,
        chunks: List[ChunkInfo]
    ) -> dict:

        result = {
            "doc_id": doc_id,
            "chunks_accepted": True,
            "alerts": [],
            "details": {}
        }

        // SP3: 跨块语义连贯性验证
        if config in [CONFIG_2_STANDARD, CONFIG_3_FULL_UPLOAD, CONFIG_5_MAX]:
            if "sp3" not in detectors:
                detectors["sp3"] = CrossChunkCoherenceVerifier(**init_params.get("sp3", {}))

            sp3_results = detectors["sp3"](chunks)
            result["details"]["sp3"] = sp3_results

            // 检查是否有任何边界异常
            anomaly_boundaries = [r for r in sp3_results if r.is_anomaly]
            if anomaly_boundaries:
                result["alerts"].append({
                    "node": "SP3",
                    "reason": f"在 {len(anomaly_boundaries)} 个 chunk 边界检测到语义断裂",
                    "details": [r.reason for r in anomaly_boundaries]
                })
                if config == CONFIG_5_MAX:
                    result["chunks_accepted"] = False

        // SP7: 语义依赖图分析（跨文档，批处理模式）
        // 通常在批次处理完成后异步执行

        return result


    // ---- 后台任务: periodic_scan ----
    // 定期扫描全库，运行 SP4
    async function periodic_scan():
        if config in [CONFIG_3_FULL_UPLOAD, CONFIG_5_MAX]:
            if "sp4" not in detectors:
                detectors["sp4"] = TriggerRegionDetector(**init_params.get("sp4", {}))

            // 从向量数据库获取全量嵌入
            all_embeddings = vector_db.fetch_all()

            sp4_results = detectors["sp4"](all_embeddings)

            // 处理高置信度异常
            high_risk = [r for r in sp4_results if r.anomaly_score > 0.7]
            for hr in high_risk:
                vector_db.quarantine(hr.doc_id)
                log_alert(f"SP4 检测到 AgentPoison 嫌疑文档: {hr.doc_id}")


    // ---- 管线入口: retrieve_and_verify ----
    // 检索 + 验证（在线查询路径）
    async function retrieve_and_verify(
        query: str,
        top_k: int = 10
    ) -> dict:

        // 1. 从向量数据库检索 Top-k
        raw_docs = vector_db.search(query, top_k=top_k)

        // SP5: 鲁棒性聚合检索
        if config in [CONFIG_4_RETRIEVAL_ONLY, CONFIG_5_MAX]:
            if "sp5" not in detectors:
                detectors["sp5"] = RobustAggregationRetriever(**init_params.get("sp5", {}))

            sp5_result = detectors["sp5"](query, raw_docs)
            return {
                "response": sp5_result["final_response"],
                "is_robust": sp5_result["is_robust"],
                "details": {"sp5": sp5_result}
            }

        // SP6: 后检索一致性验证（非鲁棒聚合模式）
        else:
            if "sp6" not in detectors:
                detectors["sp6"] = PostRetrievalVerifier(**init_params.get("sp6", {}))

            // 获取 LLM 基础回答
            llm_base = llm_model.generate(query, max_tokens=128)

            trusted, filtered, details = detectors["sp6"](
                query, raw_docs, llm_base_response=llm_base
            )

            // 使用可信文档构造上下文
            if trusted:
                context = format_context(trusted)
                final_response = llm_model.generate(
                    build_prompt(query, context)
                )
            else:
                final_response = llm_base  // 降级到无上下文回答

            return {
                "response": final_response,
                "trusted_docs": [d.doc_id for d in trusted],
                "filtered_docs": [d.doc_id for d in filtered],
                "details": details
            }


    // ---- 初始化回调 ----
    return {
        "upload_document": upload_document,
        "process_chunks": process_chunks,
        "retrieve_and_verify": retrieve_and_verify,
        "periodic_scan": periodic_scan,
        "config": config
    }
```
